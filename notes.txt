
            // copy thumbnail from shared_ns to local host for checking
            let mut thumbnail_local = shared_ns
                .walk("/data/thumbnail.ppm", Open::Read)
                .await
                .unwrap()
                .as_file()
                .unwrap();
            let mut buffer = [0; 256 * 256 * 3];
            let thumbnail_size = thumbnail_local.read(&mut buffer).await.unwrap();

            // write to images/thumbnail.ppm
            images_dir
                .create("thumbnail.ppm", Create::UserWrite, Open::ReadWrite)
                .await
                .unwrap()
                .as_file()
                .unwrap()
                .write(&buffer[..thumbnail_size])
                .await
                .unwrap();
